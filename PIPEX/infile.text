char	*get_path(char *command, char **env)
{
	t_pipex	p;
	int		i;
	int		j;

	i = 0;
	j = 0;
	while (env[i])
	{
		if (!ft_strncmp(env[i], "PATH", 4))
		{
			p.paths = get_env(env[i]);
			break ;
		}
		i++;
	}
	while (p.paths[j])
	{
		p.sub_path = ft_strjoin(p.paths[j], "/");
		p.p_exec = ft_strjoin(p.sub_path, command);
		if (!access(p.p_exec, F_OK | X_OK))
			return (p.p_exec);
		j++;
	}
	return (NULL);
}
